# Add command to makes vim automatically detect choreo file with .co suffix
echo 'au BufRead,BufNewFile *.co set filetype=co' >> your/vim_config/ftdetect/co.vim

# Then adds the syntax highlighting file
cp co.vim your/vim_config/syntax/

# I.e. to simply configure
echo 'au BufRead,BufNewFile *.co set filetype=co' >> ~/.vimrc
mkdir -p ~/.vim/syntax
cp co.vim ~/.vim/syntax/
